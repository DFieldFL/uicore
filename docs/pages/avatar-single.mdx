##Types

```js
 interface AvatarProps extends HTMLDivProps {
  name?: string
  src?: string
  backgroundColor?: string | string[] // if array it will pick a random colour from the group
  email?: string
  size?: number
  borderRadius?: number
  color?: string
  onClick?: (e: React.MouseEvent<HTMLDivElement>) => void
  fontSize?: number
}
```
