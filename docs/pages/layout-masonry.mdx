# Masonry --- docs/pages/layout-masonry.mdx

`<Layout.Masonry/>` renders a masonry layout.

## Props

```js
interface MasonryProps<T = {}> extends React.ComponentProps<typeof Container> {
  items: T[]
  renderItem: (item: T) => React.ReactElement
  keyOf: (item: T) => string | undefined
  gutter?: number
}
```

## Example

```js live=true
<Layout.Masonry
  gutter={25}
  items={[100, 200, 130, 400, 150, 100, 300, 220, 90, 250]}
  renderItem={number => (
    <Container width={190} border={{ color: Color.GREY_500 }}>
      <Text flex={{ align: 'center-center' }} height={number}>
        {number}
      </Text>
    </Container>
  )}
  keyOf={number => number}
  padding="xsmall"
/>
```
