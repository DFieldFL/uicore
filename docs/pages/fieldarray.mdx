```jsx
const fields = [
    {
      name: 'col1',
      label: 'Column Header 1',
      defaultValue: 'Item 1'
    },
    {
      name: 'col2',

      defaultValue: 'Item 2',
      renderer: value => <TextInput defaultValue={value} placeholder="Column 2 value" />
    },
    {
      name: 'col3',
      label: 'Column Header 3',
      defaultValue: 'Item 3'
    }
  ]

<FieldArray fields={...} title="Field List" noDataText="Help Text" />
```
